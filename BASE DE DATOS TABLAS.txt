create table PRODUCTS
    (
    PRODUCTID NUMBER PRIMARY KEY,
    BARCODE VARCHAR2 (13), --EAN-13
    NAME VARCHAR2 (512), --PRODUCT NAME
    DESCRIPTION VARCHAR2 (5000), --PRODUCT DESCRIPTION
    PRICE NUMBER (10,2), --PODUCT PRICE 
    CURRENCY VARCHAR2 (3), --USD/CLP/EUR 
    STOCK NUMBER, --PRODUCT QUANTITY
    CATEGORYID VARCHAR2 (255), --CATEGORY NAME (ELECTRONICS, COMPUTERS, FOOD, ETC.)
    BRAND VARCHAR2(255),-- PRODUCT BRAND
    DIMENSIONS VARCHAR2 (5000), --PRODUCT DIMENSION (WIDTH:12, HEIGHT:34, DEPTH: 25, UNIT: "MM")
    WEIGHT VARCHAR2 (5000), --PRODUCT WEIGHT EX: (VALUE:12, UNIT: "K") G,G,MG,ML,LI
    FEATURES VARCHAR2 (5000), --PRODUCT FEATURES EX: (COLOR:"RED", FORMAT: "DISPLAY")
    PROMOS VARCHAR2 (5000), --PRODUCT DISCOUNTS (VALUE: 1, UNIT:"%", START: '01/01/2000', END:...) 
    IMAGE VARCHAR2(5000 ), -- PRODUCT IMAGE (TYPE: "THUMBNAIL", URL:"BASE64:ASDSDF", DESCRIPTION:"XXX")}
    CREATED_AT DATE,
    UPDATED_AT DATE,
    OWNER VARCHAR2(255), --USER UUID, EMAIL, ETC.... REGISTRY OWNER TO
    IS_ACTIVE INTEGER
    )
    
    
    
    
    
 Vista materializada 
SELECT 
CATEGORYID AS CATEGORIA
    
FROM PRODUCTS
GROUP BY CATEGORYID
